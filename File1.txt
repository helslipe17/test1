#include <iostream>
#include <string>

using namespace std;

const int Arsize = 20;

struct car{
//    char * name[Arsize];
    string name;
    int year;
};

int main(){
    int count;
    cout << "Сколько автомобилей поместить в каталог? ";
    cin >> count;
    car *ps = new car[count];
    for(int i = 0; i < count; i++){
        cout << "Автомобиль #" << i + 1 << ":" << endl;
        cout << "Введите производителя : ";
//        cin.get(*ps[i].name, Arsize);
        cin >> ps[i].name;
        cout << "Укажите год выпуска : ";
        cin >> ps[i].year;
    }
    for(int i = 0; i < count; i++){
        cout << ps[i].year << " " << ps[i].name << endl;
    }
    cout << endl;
    return 0;
}
